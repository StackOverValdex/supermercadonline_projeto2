																																//modifica os atributos desejadoas
																																
	Context UnidadeDeMedida::eliminarUnidadeMedida(nomeUnidadeMedida:String)							//FUNÇÃO PARA REMOVER UMA UNIDADE DE MEDIDA

	Context UnidadeDeMedida::alterarEstado_Unidade(nomeUnidadeMedida:String,estadoAtividade: boolean)	//FUNÇÃO PARA ALTERAR O ESTADO DE UMA UNIDADE DE MEDIDA
